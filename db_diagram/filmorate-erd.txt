Table friends {
  id integer [pk]
  user_id integer
  friend_id integer
}

Table films {
  id integer [pk]
  name varchar
  description varchar
  release_date timestamp
  duration integer
  mpa_id integer
}

Table reviews {
  id integer [pk]
  content varchar
  is_positive bool
  user_id integer
  film_id integer
}

Table review_rates {
  id integer [pk]
  review_id integer
  user_id integer
  is_like bool
}

Table film_likes {
  id integer [pk]
  film_id integer
  user_id integer
}

Table film_genres {
  id integer [pk]
  film_id integer
  genre_id integer
}

Table users {
  id integer [pk]
  email varchar
  login varchar
  name varchar
  birthday date
}

Table genres {
  id integer [pk]
  name varchar
}

Table mpa {
  id integer [pk]
  name varchar
}

Table feeds {
  event_id integer [pk]
  timestamp timestamp
  user_id integer
  event_type varchar
  operation varchar
  entity_id integer
}
